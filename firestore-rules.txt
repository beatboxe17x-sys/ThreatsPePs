rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Products
    match /products/{id} {
      allow read: if true;
      allow write: if request.auth != null;
    }
    // Crypto addresses
    match /crypto/addresses {
      allow read: if true;
      allow write: if request.auth != null;
    }
    // Orders
    match /orders/{id} {
      allow create: if true;
      allow read: if true;
      allow update: if true;
      allow delete: if request.auth != null;
    }
    // Consent logs
    match /consentLogs/{id} {
      allow create: if true;
      allow read: if request.auth != null;
      allow update: if request.auth != null;
      allow delete: if request.auth != null;
    }
    // Visitors (live tracking)
    match /visitors/{id} {
      allow create: if true;
      allow read: if true;
      allow update: if true;
      allow delete: if request.auth != null;
    }
    // Settings (webhook URL, etc.)
    match /settings/{id} {
      allow read: if true;
      allow write: if true;
    }
  }
}
